# Устанавливаем атрибут text для всех текстовых файлов
*.txt text

# Устанавливаем атрибут text для всех файлов с расширением .md
*.md text

# Указываем Git обрабатывать .csv как текстовые файлы
*.csv text

# Для файлов с расширением .py (Python) тоже устанавливаем текст
*.py text

# Обрабатываем файлы с расширением .json как текстовые
*.json text

# Указываем формат слияния для файлов .diff (если у вас есть)
*.diff merge=union

# Принудительно устанавливаем текст для файлов, где требуется использовать LF
*.sh text eol=lf
